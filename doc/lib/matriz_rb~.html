def initialize(matriz,filas,columnas,hash_no_ceros)
   @matriz=matriz
   @filas=filas
   @columnas=columnas
   @hash_no_ceros=hash_no_ceros
end

def self.constructor(matriz)

    # if matriz.size==0 # puts “matriz vacia” # else # end

       matriz = Array.new(matriz)
   filas = matriz.size
   columnas = matriz[0].size
   hash_no_ceros={}
   n_elementos= (matriz.size * matriz[0].size)*0.6 #número de elementos que deben ser ceros para considerar la matriz como dispersa
   n_ceros=0
   filas.times do |i|
      columnas.times do |j|
         if (matriz[i][j]==0)  
            n_ceros+=1
         else
            pos_no_cero="#{i}#{j}"
            hash_no_ceros[pos_no_cero]=matriz[i][j] 
         end
      end  
   end
   if n_ceros > n_elementos
      MatrizDispersa.new(matriz,filas,columnas,hash_no_ceros)   
   else
      MatrizDensa.new(matriz,filas,columnas,hash_no_ceros)
   end
end

def +(other)
   raise ArgumentError, "La longitud de las matrices no coincide." unless self.filas == other.filas && self.columnas == other.columnas       
   sum = MatrizDensa.constructor(Array.new(filas) { Array.new(columnas) })
   self.filas.times do |i|   
      self.columnas.times do |j|
         sum.matriz[i][j] = self.matriz[i][j] + other.matriz[i][j]             
      end
   end
   return sum 
end

def -(other)
   raise ArgumentError, "La longitud de las matrices no coincide." unless @filas == other.filas && @columnas == other.columnas
   resta = MatrizDensa.constructor(Array.new(filas) { Array.new(columnas) })

      self.columnas.times do |j|
         resta.matriz[i][j] = self.matriz[i][j] - other.matriz[i][j]             
      end
   end
   return resta  #devuelve un tipo matriz modificando el objeto m1 si se hace m3=m1+m2

#Para comprobar que dos matrices son equivalentes,primero se comprueba sus dimensiones. Si tienen las mismas dimensiones se comprueba que el valor de ambas matrices sean iguales en las mismas posiciones,si esto es así se devuelve true,false en otro caso.
def ==(other)
   if ((self.filas.size==other.filas.size) && (self.columnas.size==other.columnas.size))
      self.filas.times do |i| 
         self.columnas.times do |j|
            if (self.matriz[i][j] != other.matriz[i][j])
               return false
            else
            end
         end
      end
   else
      return false
   end
   return true
end

def -@ 
   op = MatrizDensa.constructor(Array.new(filas) { Array.new(columnas) })
   self.filas.times do |i|                   
      self.columnas.times do |j|
         op.matriz[i][j] = -self.matriz[i][j]
      end
   end
   return op

end

def Producto_escalar (other)
   mul = MatrizDensa.constructor(Array.new(filas) { Array.new(columnas) })
   self.filas.times do |i|  
      self.columnas.times do |j|
         mul.matriz[i][j] = self.matriz[i][j] * other
      end
   end
   return mul
end

#Dos matrices son multiplicables si el numero de columnas de A coincide con el numero de filas de B
def * (other)
   if other.instance_of? Fixnum
      return (Producto_escalar(other))
   else
      raise ArgumentError, "La longitud de las matrices no coincide." unless @columnas == other.filas
      elemento = Array.new
      acumulado = 0
      self.filas.times do |i|
         elemento_fila = Array.new
         other.columnas.times do |j|
            acumulado = 0
            self.columnas.times do |k|
               suma = @matriz[i][k] * other.matriz[k][j]
               acumulado = suma + acumulado
            end
            elemento_fila << acumulado
         end
         elemento << elemento_fila
      end
   end
   MatrizDensa.constructor(elemento)
end
